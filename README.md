## Install

```bash
npm install --save react-faq
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-faq'
import 'react-faq/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
